Lemma posroot_odd :
  ∀ x₁ y₁ r θ
         (rgt0 : r ≠ 0)
         (phase : straight r 0 0 0 x₁ y₁),
    (∃ (n:Z), κ₂ x₁ y₁ r θ = θ + IZR (2 × n + 1) × PI) →
    (¬ (2 × r - y₁ = 0) ∧
    ∃ (m:Z),
      θ = 2 × atan ((x₁ + sqrt (x₁² - (2 × r - y₁) × y₁))/(2 × r - y₁)) +
          IZR (2 × m) × PI) ∨
    ((2 × r - y₁ = 0) ∧ 0 < x₁ ∧
     ∃ (m:Z),
       θ = IZR (2 × m + 1) × PI) ∨
    ((2 × r - y₁ = 0) ∧ x₁ < 0 ∧
     ∃ (m:Z),
       θ = 2 × atan (r/x₁) +
           IZR (2 × m) × PI).

Lemma negroot_even :
  ∀ x₁ y₁ r θ
         (rgt0 : r ≠ 0)
         (phase : straight r 0 0 0 x₁ y₁),
    (∃ (n:Z), κ₂ x₁ y₁ r θ = θ + IZR (2 × n) × PI) →
    (¬ (2 × r - y₁ = 0) ∧
    ∃ (m:Z),
      θ = 2 × atan ((x₁ - sqrt (x₁² - (2 × r - y₁) × y₁))/(2 × r - y₁)) +
          IZR (2 × m) × PI) ∨
    ((2 × r - y₁ = 0) ∧ 0 < x₁ ∧
     ∃ (m:Z),
       θ = 2 × atan (r / x₁) +
           IZR (2 × m) × PI) ∨
    ((2 × r - y₁ = 0) ∧ x₁ < 0 ∧
     ∃ (m:Z),
       θ = IZR (2 × m + 1) × PI).

Lemma theta2_odd :

         (phase : straight r 0 0 0 x₁ y₁),
    (∃ (n:Z), κ₂ x₁ y₁ r θ = θ + IZR (2 × n + 1) × PI) →
    (∃ (m:Z), θ = θ2 x₁ y₁ r + IZR (2 × m) × PI).

Lemma theta1_even :
  ∀ x₁ y₁ r θ
         (rne0 : r ≠ 0)
         (phase : straight r 0 0 0 x₁ y₁),
    (∃ (n:Z), κ₂ x₁ y₁ r θ = θ + IZR (2 × n) × PI) →
    (∃ (m:Z), θ = θ1 x₁ y₁ r + IZR (2 × m) × PI).

Lemma theta1_ne_theta2 :
  ∀ x y r n (rne0 : r ≠ 0)
    (phase : straight r 0 0 0 x y),
    θ1 x y r ≠ θ2 x y r + IZR (2 × n) × PI.

Lemma k2_theta1_even :
  ∀ x y r
         (rne0 : r ≠ 0)
         (phase : straight r 0 0 0 x y),
    ∃ q, κ₂ x y r (θ1 x y r) = (θ1 x y r) + 2 × IZR q × PI.

Lemma k2_theta2_odd :

    (∃ n : Z, κ₂ x y r (θ2 x y r) =
                   (θ2 x y r) + IZR (2 × n + 1) × PI).
