Lemma minlength_outer_infinite_hat_straight_std :
  ∀ r x y φ₂
         (p1p2 : 0 < φ₂)
         (p2ub : φ₂ ≤ 2 × PI)
         (lt : 0 < r)
         (oc : 2 × r × y < x² + y²),
    let wx := r×sin φ₂ in
    let wy := r*(1 - cos φ₂) in
    (y ≥ 0 ∧ wx × y ≤ wy × x) →
    sqrt (x² + y²) ≤ L x y (θ1 x y r) r.

Lemma maxlength_outer_infinite_hat_straight_std :
  ∀ r x y φ₂
         (p1p2 : 0 < φ₂)
         (p2ub : φ₂ < 2 × PI)
         (lt : 0 < r)
         (oc : 2 × r × y < x² + y²),

    let wy := r*(1 - cos φ₂) in
