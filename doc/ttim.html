Library ttim

Require Import Reals.
Require Import Coquelicot.Coquelicot.
Require Import Lra.
Require Import ttyp.
Import Lra.


Record arrivebounds (p:pt) :=
  mktm { te : R ; tl : R ; boundorder : te ≤ tl}.

Definition te' {p:pt} (b:arrivebounds p) := te p b.
Definition tl' {p:pt} (b:arrivebounds p) := tl p b.

Definition tarrive {p:pt} (b : arrivebounds p) (t : R) :=
  te' b ≤ t ≤ tl' b.

Definition collision_possibility {p:pt} (o i : arrivebounds p) (t : R) :=
  tarrive o t ∧ tarrive i t.

Definition Wie {p:pt} (i o : arrivebounds p):= te' o ≤ te' i ≤ tl' o.
Definition Wil {p:pt} (i o : arrivebounds p):= te' o ≤ tl' i ≤ tl' o.
Definition Woe {p:pt} (i o : arrivebounds p):= te' i ≤ te' o ≤ tl' i.
Definition Wol {p:pt} (i o : arrivebounds p):= te' i ≤ tl' o ≤ tl' i.

Definition We {p:pt} (i o : arrivebounds p) := Wie i o ∨ Woe i o.
Definition Wl {p:pt} (i o : arrivebounds p) := Wil i o ∨ Wol i o.

Lemma Wil_possible_collision : ∀ {p:pt} (o i : arrivebounds p),
    Wil i o → ∃ t, collision_possibility o i t.

Lemma Wie_possible_collision : ∀ {p:pt} (o i : arrivebounds p),
    Wie i o → ∃ t, collision_possibility o i t.

Lemma Wol_possible_collision : ∀ {p:pt} (o i : arrivebounds p),
    Wol i o → ∃ t, collision_possibility o i t.

Lemma Woe_possible_collision : ∀ {p:pt} (o i : arrivebounds p),
    Woe i o → ∃ t, collision_possibility o i t.

Lemma We_possible_collision : ∀ {p:pt} (o i : arrivebounds p),
    We i o ↔ ∃ t, collision_possibility o i t.

Lemma Wl_possible_collision : ∀ {p:pt} (o i : arrivebounds p),
    Wl i o ↔ ∃ t, collision_possibility o i t.

Lemma W_possible_collision : ∀ {p:pt} (o i : arrivebounds p),
    We i o ↔ Wl i o.

Definition W {p:pt} (o i: arrivebounds p) := We o i.
